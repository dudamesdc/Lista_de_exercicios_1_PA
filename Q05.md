  printf(" &vet[%d] = %X",i, &vet[i]); // Imprime o endereço de array vet[i].
  printf(" (f + %d) = %X",i, f+i); // Imprime o endereço calculado pelo ponteiro f somado ao índice i.
  printf("\n");
  }
}
